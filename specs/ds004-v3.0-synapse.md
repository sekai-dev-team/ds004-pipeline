# DS-004 Synapse Architecture — Implementation Spec

> v3.0: SYNAPSE-based episodic-semantic consolidation.
> Replaces Karpathy LLM Wiki v2.1 entirely.

## Core Change

```
v2.1:  Article → k-mcp search → frontmatter parse → LLM per note
v3.0:  Article → LLM tag (DS-001) → group by tag → LLM per tag (DS-004)
```

No more k-mcp search for concept lookup. Tags ARE the concept keys.

### Post-processing

```python
def _process_tags(concept_tags, tag_library):
    for tag in concept_tags:
        if tag["name"] in tag_library:
            tag_library[tag["name"]]["article_count"] += 1
            tag_library[tag["name"]]["last_seen"] = utcnow()
        else:
            # Embed new tag's label_text via k-mcp embed endpoint
            embedding = kmcp_client.embed(tag["label_text"])
            # Check cosine similarity against all library tags
            best_match, best_score = _find_best_match(embedding, tag_library)
            if best_score > 0.90:
                # Reuse existing tag (semantic duplicate)
                tag["name"] = best_match
                tag_library[best_match]["article_count"] += 1
            else:
                # Create new tag
                _write_tag_note(tag["name"], tag["label_text"], embedding)
                tag_library[tag["name"]] = {
                    "name": tag["name"],
                    "label_text": tag["label_text"],
                    "embedding": embedding,
                    "article_count": 1,
                    "concept_page": None,
                    "first_seen": utcnow(),
                    "last_seen": utcnow()
                }
```

## Phase 3: DS-004 Rewrite

### New `consolidator.py` Flow

```python
def consolidate_all():
    # Step 1: Find new episodic notes (same as v2.1)
    new_notes = _find_new_episodic_notes()
    
    # Step 2: Group by topic/ tags
    tag_groups = {}
    for note_path in new_notes:
        content = read_vault_file(note_path)
        fm = _parse_frontmatter(content)
        tags = fm.get("tags", [])
        topics = [t.replace("topic/", "") for t in tags if t.startswith("topic/")]
        for topic in topics:
            tag_groups.setdefault(topic, []).append(note_path)
    
    # Step 3: For each tag, decide action
    tag_library = read_tag_library()
    consolidation_tasks = []
    for tag_name, note_paths in tag_groups.items():
        tag_info = tag_library.get(tag_name, {})
        concept_page = tag_info.get("concept_page")
        article_count = tag_info.get("article_count", len(note_paths))
        
        if concept_page:
            # UPDATE: existing concept page
            concept_content = read_vault_file(concept_page)
            task = ("update", tag_name, note_paths, concept_content)
        elif article_count >= 2:  # θ=2
            # CREATE: new concept page
            task = ("create", tag_name, note_paths, None)
        else:
            # SKIP: not enough articles
            task = ("skip", tag_name, note_paths, None)
        consolidation_tasks.append(task)
    
    # Step 4: Serial LLM calls (one per concept)
    for action, tag_name, note_paths, concept_content in consolidation_tasks:
        if action == "skip":
            continue
        
        prompt = _build_prompt(action, tag_name, note_paths, concept_content)
        llm_response = LLM(prompt)
        
        if action == "create":
            write_vault_file(f"concepts/{tag_name}.md", llm_response)
            update_tag_note(tag_name, {"concept_page": f"concepts/{tag_name}.md"})
        else:
            write_vault_file(f"concepts/{tag_name}.md", llm_response)
    
    # Step 5: Inject wikilinks into source notes
    for note_path in new_notes:
        content = read_vault_file(note_path)
        fm = _parse_frontmatter(content)
        topics = [t.replace("topic/", "") for t in fm.get("tags", []) if t.startswith("topic/")]
        for topic in topics:
            tag_info = tag_library.get(topic, {})
            if tag_info.get("concept_page"):
                _inject_wikilink(note_path, f"concepts/{topic}.md")
    
    # Step 6: Cross-concept association edges (post-processing)

    # Inject [[wikilinks]] in "## 相关概念" section
    
    # Step 7: Update index.md + log.md (same as v2.1)
```

```python
def embed(text: str) -> list[float] | None:
    """Generate embedding vector for text via k-mcp."""
    result = _call_tool("embed", {"text": text})
    if result is None:
        return None
    # Parse MCP content format
    content = result.get("content", [])
    if isinstance(content, list) and content:
        text_content = content[0].get("text", "[]")
        return json.loads(text_content)
    return None
```

### Cosine Similarity Helper

```python
def cosine_similarity(a: list[float], b: list[float]) -> float:
    dot = sum(x * y for x, y in zip(a, b))
    norm_a = math.sqrt(sum(x * x for x in a))
    norm_b = math.sqrt(sum(x * x for x in b))
    return dot / (norm_a * norm_b) if norm_a and norm_b else 0.0
```

## Container Constraint

ALL embedding must go through k-mcp container. DS-004 container does NOT load ONNX models. 
The `embed` MCP tool in k-mcp is the single embedding provider.
